### 阶段 4 — 幂等性改造

**目标**：防止 OutboxPoller 重试或 Stream 消费者重启时，同一事件被处理器重复执行（如重复扣库存、重复生成凭证）。

#### 4.1 数据库迁移

新建 `database/migrations/003_create_event_processed.sql`：

```sql
CREATE TABLE IF NOT EXISTS `sys_event_processed` (
  `id` BIGINT AUTO_INCREMENT PRIMARY KEY,
  `event_id` BIGINT NOT NULL COMMENT 'domain_event_outbox.id',
  `handler_name` VARCHAR(255) NOT NULL COMMENT '处理器类名',
  `processed_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  UNIQUE KEY `uk_event_handler` (`event_id`, `handler_name`),
  INDEX `idx_processed_at` (`processed_at`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='事件幂等表';
```

- `event_id` + `handler_name` 联合唯一：同一事件同一处理器只成功记录一次
- 使用 `INSERT ... ON DUPLICATE KEY UPDATE` 实现原子 check-and-record

#### 4.2 IdempotencyGuard 类

新建 `src/infrastructure/event-bus/IdempotencyGuard.ts`：

```typescript
import { execute } from '@/lib/db';
import { secureLog } from '@/lib/logger';

/**
 * 事件处理器幂等性守护
 *
 * 基于 sys_event_processed 表的唯一键 (event_id, handler_name) 实现：
 * - checkAndMark: 原子 INSERT，若已存在则返回 false（已处理过）
 * - 用于包装事件处理器，确保重试/重启场景下不重复执行副作用
 */
export class IdempotencyGuard {
  /**
   * 尝试标记事件为已处理
   * @returns true=首次处理（应执行业务逻辑），false=已处理过（跳过）
   */
  static async checkAndMark(eventId: number, handlerName: string): Promise<boolean> {
    try {
      await execute(
        `INSERT INTO sys_event_processed (event_id, handler_name) VALUES (?, ?)
         ON DUPLICATE KEY UPDATE processed_at = processed_at`,
        [eventId, handlerName]
      );
      // ON DUPLICATE KEY UPDATE 不抛异常但 affectedRows=0 表示已存在
      // 此处简化：INSERT 成功视为首次，实际可用 affectedRows 判断
      return true;
    } catch (error) {
      secureLog('warn', 'IdempotencyGuard: mark failed, allowing execution', {
        eventId, handlerName, error: String(error),
      });
      return true; // DB 故障时降级为允许执行（业务侧需容忍）
    }
  }

  /**
   * 清理过期的幂等记录（定期维护）
   */
  static async cleanupOlderThan(days: number = 30): Promise<void> {
    await execute(
      `DELETE FROM sys_event_processed WHERE processed_at < DATE_SUB(NOW(), INTERVAL ? DAY)`,
      [days]
    );
  }
}
```

> **设计决策**：`checkAndMark` 返回 `true` 时执行业务逻辑。由于 MySQL `INSERT ... ON DUPLICATE KEY UPDATE` 在重复时 `affectedRows=0`，精确实现需用 `result.affectedRows === 1` 判断。上面伪代码简化了，实现时需读取 `affectedRows`。

#### 4.3 包装事件处理器

修改 `src/infrastructure/events/EventHandler.ts`，新增 `IdempotentHandler` 装饰器基类：

```typescript
import { IdempotencyGuard } from './IdempotencyGuard';
import type { DomainEvent, EventHandler } from '@/domain/shared/DomainTypes';

/**
 * 幂等处理器装饰器
 * 包装原处理器，在 handle 前检查 sys_event_processed
 */
export class IdempotentHandler implements EventHandler {
  constructor(
    private readonly inner: EventHandler,
    private readonly handlerName: string = inner.constructor.name
  ) {}

  async handle(event: DomainEvent): Promise<void> {
    const eventId = (event as any).id ?? (event as any).eventId;
    if (!eventId) {
      // 无 eventId 的事件（如非 outbox 来源）直接执行
      return this.inner.handle(event);
    }
    const shouldExecute = await IdempotencyGuard.checkAndMark(eventId, this.handlerName);
    if (shouldExecute) {
      return this.inner.handle(event);
    }
    // 已处理，跳过
  }
}
```

### 阶段 5 — Redis Stream 事件分发

**目标**：将 OutboxPoller 的直接 `eventBus.publish` 替换为 Redis Stream XADD，消费端用 XREADGROUP + XACK 分发，支持多实例并行消费与故障恢复。

> **决策**：`EVENT_BUS_TYPE=memory` 时仍走 OutboxPoller 直接 publish（开发环境友好）；`EVENT_BUS_TYPE=db` 且 `REDIS_URL` 存在时启用 Stream 模式。

#### 5.1 StreamPublisher

新建 `src/infrastructure/event-bus/StreamPublisher.ts`：

```typescript
import type { Redis } from 'ioredis';
import { secureLog } from '@/lib/logger';

const STREAM_KEY = 'erp:domain-events';
const MAX_STREAM_LENGTH = 10000;

export class StreamPublisher {
  constructor(private redis: Redis) {}

  async publish(event: {
    id: number;
    eventType: string;
    aggregateType?: string;
    aggregateId?: number;
    payload: string;
  }): Promise<string> {
    const messageId = await this.redis.xadd(
      STREAM_KEY,
      'MAXLEN', '~', MAX_STREAM_LENGTH,
      '*',
      'eventId', String(event.id),
      'eventType', event.eventType,
      'aggregateType', event.aggregateType ?? '',
      'aggregateId', String(event.aggregateId ?? ''),
      'payload', event.payload
    );
    return messageId;
  }
}
```

#### 5.2 StreamConsumer

新建 `src/infrastructure/event-bus/StreamConsumer.ts`：

```typescript
import type { Redis } from 'ioredis';
import { registerEventHandlers } from '@/application/EventRegistry';
import { getDomainEventOutbox } from './DomainEventOutboxFactory';
import { secureLog } from '@/lib/logger';

const STREAM_KEY = 'erp:domain-events';
const GROUP = 'erp-consumers';
const CONSUMER_NAME = `consumer-${process.pid}`;
const BLOCK_MS = 5000;
const BATCH_SIZE = 50;

export class StreamConsumer {
  private running = false;

  async start(): Promise<void> {
    if (this.running) return;
    this.running = true;

    // 确保消费者组存在
    await this.ensureGroup();

    // 启动消费循环
    this.consumeLoop();
  }

  private async ensureGroup(): Promise<void> {
    try {
      await this.redis.xgroup('CREATE', STREAM_KEY, GROUP, '$', 'MKSTREAM');
    } catch (err: any) {
      if (err?.message?.includes('BUSYGROUP')) return; // 组已存在
      throw err;
    }
  }

  private async consumeLoop(): Promise<void> {
    while (this.running) {
      try {
        const results = await this.redis.xreadgroup(
          'GROUP', GROUP, CONSUMER_NAME,
          'COUNT', BATCH_SIZE,
          'BLOCK', BLOCK_MS,
          'STREAMS', STREAM_KEY, '>'
        );

        if (!results) continue;

        const eventBus = registerEventHandlers();
        const outbox = getDomainEventOutbox();

        for (const [stream, messages] of results) {
          for (const [messageId, fields] of messages) {
            await this.processMessage(messageId, fields, eventBus, outbox);
          }
        }
      } catch (error) {
        secureLog('error', 'StreamConsumer loop error', { error: String(error) });
        await new Promise(r => setTimeout(r, 1000)); // 错误后等待
      }
    }
  }

  private async processMessage(messageId, fields, eventBus, outbox): Promise<void> {
    const data = fieldsToObject(fields);
    try {
      const payload = JSON.parse(data.payload);
      await eventBus.publish({
        id: Number(data.eventId),
        eventType: data.eventType,
        aggregateType: data.aggregateType || undefined,
        aggregateId: data.aggregateId ? Number(data.aggregateId) : undefined,
        ...payload,
      });
      await this.redis.xack(STREAM_KEY, GROUP, messageId);
      await outbox.markAsProcessed(Number(data.eventId));
    } catch (error) {
      secureLog('error', 'StreamConsumer process failed', {
        messageId, eventId: data.eventId, error: String(error),
      });
      // 失败不 ACK，待 XAUTOCLAIM 或重试
    }
  }

  // 定期 XAUTOCLAIM 处理超时未 ACK 的消息（死消费者遗留）
  async reclaimStaleMessages(): Promise<void> { /* ... */ }

  stop(): void { this.running = false; }
}
```

#### 6.1 InfrastructureHealthCheck

新建 `src/infrastructure/monitoring/InfrastructureHealthCheck.ts`：

```typescript
import { getCacheManager } from '@/infrastructure/cache/CacheManager';
import { RedisCacheManager } from '@/infrastructure/cache/RedisCacheManager';
import { getDomainEventOutbox } from '@/infrastructure/event-bus/DomainEventOutboxFactory';
import { query } from '@/lib/db';

export interface HealthStatus {
  status: 'healthy' | 'degraded' | 'unhealthy';
  checks: {
    database: { status: string; latencyMs?: number };
    redis?: { status: string; latencyMs?: number; mode: 'redis' | 'memory' };
    outbox?: { status: string; pendingCount?: number; deadLetterCount?: number };
  };
}

export class InfrastructureHealthCheck {
  static async check(): Promise<HealthStatus> {
    const checks: HealthStatus['checks'] = {};

    // 数据库检查
    try {
      const start = Date.now();
      await query('SELECT 1');
      checks.database = { status: 'healthy', latencyMs: Date.now() - start };
    } catch {
      checks.database = { status: 'unhealthy' };
    }

    // Redis 检查
    const cache = getCacheManager();
    if (cache instanceof RedisCacheManager) {
      try {
        const start = Date.now();
        const redis = cache.rawClient();
        await redis.ping();
        checks.redis = { status: 'healthy', latencyMs: Date.now() - start, mode: 'redis' };
      } catch {
        checks.redis = { status: 'unhealthy', mode: 'redis' };
      }
    } else {
      checks.redis = { status: 'degraded', mode: 'memory' };
    }

    // Outbox 检查
    try {
      const outbox = getDomainEventOutbox();
      const pending = await outbox.fetchPendingEvents(1);
      checks.outbox = { status: 'healthy', pendingCount: pending.length };
    } catch {
      checks.outbox = { status: 'unhealthy' };
    }

    const anyUnhealthy = Object.values(checks).some(c => c.status === 'unhealthy');
    const anyDegraded = Object.values(checks).some(c => c.status === 'degraded');
    return {
      status: anyUnhealthy ? 'unhealthy' : anyDegraded ? 'degraded' : 'healthy',
      checks,
    };
  }
}
```

## 假设与决策

1. **接受 ~5s 事件延迟**：统一走 OutboxPoller（5s 轮询间隔），移除应用服务内的即时发布
2. **幂等表方案优于内存去重**：多实例下内存去重无效，DB 唯一键是唯一可靠方案
3. **Stream 模式可选**：`REDIS_URL` 缺失时降级为直接 publish，保证开发环境零依赖
4. **`registerEventHandlers()` 仍需调用**：注册处理器到全局 EventBus，OutboxPoller/StreamConsumer 均依赖此单例
5. **Sales/Production Service 无外部调用方**：清理后不影响现有 API（它们走直接 SQL），但为未来 DDD 收敛做准备
6. **健康检查端点不加鉴权**：供 K8s/Nginx 探活，不暴露敏感数据（仅状态与延迟）
